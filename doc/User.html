def has_roles?(role)
  role.each do |r|
    if !roles.include?(r)
      return false
    end
  end
  return true
end

def roles=(roles)
  roles = [*roles].map { |r| r.to_sym }
  self.roles_mask = (roles & ROLES).map { |r| 2**ROLES.index(r) }.inject(0, :+)
end
